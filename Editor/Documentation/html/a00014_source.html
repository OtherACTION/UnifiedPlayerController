641 private void FirstPersonMove()
642 {
643 float targetSpeed = _input.sprint ? FP_SprintSpeed : FP_MoveSpeed;
644 if (_input.move == Vector2.zero) targetSpeed = 0.0f;
645
646 float speedOffset = 0.1f;
647 float inputMagnitude = _input.analogMovement ? _input.move.magnitude : 1f;
648
649 // Smoothly adjust speed to target
650 if (Mathf.Abs(_speed - targetSpeed * inputMagnitude) > speedOffset)
651 {
652 _speed = Mathf.MoveTowards(_speed, targetSpeed * inputMagnitude, FP_SpeedChangeRate * Time.deltaTime);
653 }
654 else
655 {
656 _speed = targetSpeed * inputMagnitude;
657 }
658
659 // Blend animation for smooth transitions
660 _animationBlend = Mathf.Lerp(_animationBlend, targetSpeed, Time.deltaTime * FP_SpeedChangeRate);
661 if (_animationBlend < 0.01f) _animationBlend = 0f;
662
663 // Combine input with player orientation
664 Vector3 inputDirection = (transform.right * _input.move.x + transform.forward * _input.move.y);
665 if (inputDirection.sqrMagnitude > 0.01f)
666 inputDirection = inputDirection.normalized;
667 else
668 inputDirection = Vector3.zero;
669
670 // Move the character controller
671 _controller.Move(inputDirection * (_speed * inputMagnitude * Time.deltaTime) +
672 new Vector3(0.0f, _verticalVelocity, 0.0f) * Time.deltaTime);
673
674 // Update animator parameters if present
675 if (_hasAnimator)
676 {
677 float normalizedBlend = 0f;
678 if (_animationBlend > 0f)
679 {
680 if (_animationBlend <= TP_MoveSpeed)
681 normalizedBlend = Mathf.InverseLerp(0f, TP_MoveSpeed, _animationBlend) * 0.5f;
682 else
683 normalizedBlend = 0.5f + Mathf.InverseLerp(TP_MoveSpeed, TP_SprintSpeed, _animationBlend) * 0.5f;
684 }
685 normalizedBlend = Mathf.Round(normalizedBlend * 100f) / 100f;
686 _animator.SetFloat("Speed", normalizedBlend);
687
688 // Use the actual input value for smooth blending
689 _animator.SetFloat("Direction", _input.move.y);
690 _animator.SetFloat(_animIDMotionSpeed, inputMagnitude);
691 }
692 }

725 private void JumpAndGravity()
726 {
727 if (Grounded)
728 {
729 // reset the fall timeout timer
730 _fallTimeoutDelta = FallTimeout;
731
732 // update animator if using character
733 if (_hasAnimator)
734 {
735 _animator.SetBool(_animIDJump, false);
736 _animator.SetBool(_animIDFreeFall, false);
737 }
738
739 // stop our velocity dropping infinitely when grounded
740 if (_verticalVelocity < 0.0f)
741 {
742 _verticalVelocity = -2f;
743 }
744
745 // Jump
746 if (_input.jump && _jumpTimeoutDelta <= 0.0f)
747 {
748 // the square root of H * -2 * G = how much velocity needed to reach desired height
749 _verticalVelocity = Mathf.Sqrt(JumpHeight * -2f * Gravity);
750
751 // update animator if using character
752 if (_hasAnimator)
753 {
754 _animator.SetBool(_animIDJump, true);
755 }
756 }
757
758 // jump timeout
759 if (_jumpTimeoutDelta >= 0.0f)
760 {
761 _jumpTimeoutDelta -= Time.deltaTime;
762 }
763 }
764 else
765 {
766 // reset the jump timeout timer
767 _jumpTimeoutDelta = JumpTimeout;
768
769 // fall timeout
770 if (_fallTimeoutDelta >= 0.0f)
771 {
772 _fallTimeoutDelta -= Time.deltaTime;
773 }
774 else
775 {
776 // update animator if using character
777 if (_hasAnimator)
778 {
779 _animator.SetBool(_animIDFreeFall, true);
780 }
781 }
782
783 // if we are not grounded, do not jump
784 _input.jump = false;
785 }
786
787 // apply gravity over time if under terminal (multiply by delta time twice to linearly speed up over time)
788 if (_verticalVelocity < _terminalVelocity)
789 {
790 _verticalVelocity += Gravity * Time.deltaTime;
791 }
792 }
